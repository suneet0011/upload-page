from pyspark.sql.functions import trim, when, col

df = spark.read.csv("WA_Fn-UseC_-Telco-Customer-Churn.csv", header=True, inferSchema=True)

# Replace blank strings with null and cast safely
df = df.withColumn("TotalCharges_cleaned", when(trim(col("TotalCharges")) == "", None).otherwise(col("TotalCharges")))
df = df.withColumn("TotalCharges_cleaned", col("TotalCharges_cleaned").cast("double"))

# Drop rows with null TotalCharges
df = df.na.drop(subset=["TotalCharges_cleaned"])

# Replace original column
df = df.drop("TotalCharges").withColumnRenamed("TotalCharges_cleaned", "TotalCharges")

# Convert SeniorCitizen 0/1 to Yes/No
df = df.withColumn("SeniorCitizen", when(col("SeniorCitizen") == 1, "Yes").otherwise("No"))

from pyspark.ml.feature import StringIndexer, OneHotEncoder, VectorAssembler
from pyspark.ml import Pipeline

# Categorical features
cat_features = ['gender','SeniorCitizen','Partner','Dependents','PhoneService',
                'MultipleLines','InternetService','OnlineSecurity','OnlineBackup',
                'DeviceProtection','TechSupport','StreamingTV','StreamingMovies',
                'Contract','PaperlessBilling','PaymentMethod']

# Index + Encode
indexers = [StringIndexer(inputCol=c, outputCol=c+"_Index", handleInvalid='skip') for c in cat_features]
encoders = [OneHotEncoder(inputCol=c+"_Index", outputCol=c+"_Vec") for c in cat_features]

# Label indexer
label_indexer = StringIndexer(inputCol="Churn", outputCol="label", handleInvalid='skip')

# VectorAssembler
assembler_inputs = [c+"_Vec" for c in cat_features] + ['tenure', 'MonthlyCharges', 'TotalCharges']
assembler = VectorAssembler(inputCols=assembler_inputs, outputCol="features")

# Full pipeline
pipeline = Pipeline(stages=indexers + encoders + [label_indexer, assembler])
df_prepared = pipeline.fit(df).transform(df)

from pyspark.sql.functions import udf
from pyspark.sql.types import StringType

# Define action based on churn probability
def retention_recommendation(prob):
    prob_churn = prob[1]
    if prob_churn > 0.7:
        return "Send Offer"
    elif prob_churn > 0.4:
        return "Monitor Closely"
    else:
        return "No Action Needed"

action_udf = udf(retention_recommendation, StringType())

# Apply to predictions
predictions_with_action = predictions.withColumn("Retention_Action", action_udf(col("probability")))
predictions_with_action.select("probability", "prediction", "Retention_Action").show(10, truncate=False)
# Count action types
action_counts = predictions_with_action.groupBy("Retention_Action").count().toPandas()

# Plot pie chart
import matplotlib.pyplot as plt

plt.figure(figsize=(6, 6))
plt.pie(action_counts["count"], labels=action_counts["Retention_Action"], autopct='%1.1f%%', startangle=140)
plt.title("Recommended Retention Actions Based on Churn Probability")
plt.tight_layout()
plt.show()
